select distinct stops.stop_id, stops.stop_name, stops.stop_desc, trips.route_id into outfile '/var/lib/mysql-files/data_distinct.csv' fields terminated by ',' lines terminated by '\n' FROM stops, trips, times where stops.stop_id = times.stop_id and times.trip_id = trips.trip_id;

select distinct timing.stop_id, timing.arrival_time, trips.route_id into outfile '/var/lib/mysql-files/timings.csv' fields terminated by ',' lines terminated by '\n' FROM trips, timing where trips.trip_id = timing.trip_id;
